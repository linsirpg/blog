简单-226. 翻转二叉树

linsir 2020-04-20 力扣-简单

# 题目描述

翻转一棵二叉树。

# 例子

输入：
     4
   /   \
  2     7
 / \   / \
1   3 6   9
输出：
     4
   /   \
  7     2
 / \   / \
9   6 3   1

# 解题方法

/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @return {TreeNode}
 */
var invertTree = function (root) {
  if (!root) { return root; }
  if (root.left != null || root.right != null) {
    var leftNode = root.left;
    var rightNode = root.right;
    root.right = leftNode;
    root.left = rightNode;
    invertTree(root.left);
    invertTree(root.right);
  }
  return root;
};

# 解题思路

1.递归调用,当当前节点存在子节点时,翻转左右节点

var invertTree = function (root) {
  if (!root) { return root; }
  if (root.left != null || root.right != null) {
    //获取左右节点
    var leftNode = root.left;
    var rightNode = root.right;
    //翻转左右节点
    root.right = leftNode;
    root.left = rightNode;
    //递归看给节点是否需要翻转
    invertTree(root.left);
    invertTree(root.right);
  }
  return root;
};
