# 排序合并两个正序数组

输入 [3, 5, 7] [1, 2, 9] 结果 [1, 2, 3, 5, 7, 9]

function merge(left, right) {
  var res = [];
  var leftLen = left.length;
  var rightLen = right.length;
  var len = left.length + right.length;
  for (var index = 0, i = 0, j = 0; index < len; index++) {
    if (i >= leftLen) {
      res[index] = right[j]
      j++
    } else if (j >= rightLen) {
      res[index] = left[i]
      i++
    } else if (left[i] > right[j]) {
      res[index] = right[j]
      j++
    } else if (left[i] <= right[j]) {
      res[index] = left[i]
      i++
    }
  }
  console.log(res)
}
merge([3, 5, 7], [1, 2, 9])
